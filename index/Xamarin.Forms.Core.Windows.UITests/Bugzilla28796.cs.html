	[Issue (IssueTracker.Bugzilla, 28796, "Crash on Tab change", PlatformAffected.Android, NavigationBehavior.PushModalAsync)]
	public class Bugzilla28796
		: TabbedPage
	{

			Children.Add (new ContentPage {
