	internal static class Rects
	{
		public static AppRect RootViewRect (this IApp app)
		{
#if __WINDOWS__
			return app.Query("Xamarin.Forms.ControlGallery.WindowsUniversal")[0].Rect;
#else
			return app.Query (q => q.Raw ("* index:0"))[0].Rect;
#endif
		}
