			public void TestDynamicResources (bool useCompiledXaml)
			{
				var layout = new DynamicResource (useCompiledXaml);
				var label = layout.label0;
 
				Assert.Null (label.Text);
 
				layout.Resources = new ResourceDictionary { 
					{"FooBar", "FOOBAR"},
